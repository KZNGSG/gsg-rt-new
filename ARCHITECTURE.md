## 🎯 ГЛАВНАЯ ИДЕЯ

Каждый ТР ТС = отдельная страница с уникальным контентом.
Разделяем по ТИПУ ДОКУМЕНТА (сертификат vs декларация), а не просто по регламенту.

**Почему это работает:**
- Пользователь ищет "сертификат на игрушки" → попадает на нашу страницу
- Пользователь ищет "декларация на одежду" → попадает на нашу страницу
- 50+ целевых страниц × 1000+ запросов = монополия в поиске

```typescript
export interface TRTSRegulation {
  id: string;                    // "007-2011"
  number: string;                // "ТР ТС 007/2011"
  slug: string;                  // "007-detskie-tovary"
  name: string;                  // "О безопасности продукции для детей"
  shortName: string;             // "Детские товары"

  // Типы документов
  documentTypes: {
    certificate: boolean;        // Нужен сертификат?
    declaration: boolean;        // Нужна декларация?
    sgr: boolean;               // Нужно СГР?
    sgrCategories?: string[];   // Для каких категорий нужно СГР
  };

  // SEO
  seo: {
    title: string;
    description: string;
    keywords: string[];
    h1: string;
  };

  // Контент для страницы
  content: {
    description: string;         // Описание регламента
    products: string[];          // Какие товары подлежат
    documents: string[];         // Необходимые документы
    process: string[];           // Этапы оформления
    faq: { q: string; a: string }[];
  };

  // Цены и сроки
  pricing: {
    serial: { price: number; days: string };      // Серийное
    batch: { price: number; days: string };       // Партия
    urgent: { price: number; days: string };      // Срочное
  };

  // Схемы подтверждения
  schemes: string[];             // ["1с", "2с", "3с"] или ["1д", "2д", "3д"]

  // Срок действия документа
  validity: string;              // "до 5 лет" или "на партию"

  // Связанные регламенты
  relatedTRTS: string[];         // ["020-2011", "004-2011"]

  // Связанные ТН ВЭД коды
  tnvedCodes: string[];          // ["9503", "9504"]
}

export const TR_TS_DATABASE: Record<string, TRTSRegulation> = {
  // ... все регламенты
};
```
